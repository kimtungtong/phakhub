require('dotenv').config() const express = require('express') const path = require('path') const { createClient } = require('@supabase/supabase-js') const app = express() app.use(express.json()) // บังคับให้เซิร์ฟเวอร์ Vercel คลายล็อกการอ่านไฟล์หน้าบ้านในโฟลเดอร์ public ทั้งหมด app.use(express.static(path.join(__dirname, 'public'))) // เชื่อมต่อฐานข้อมูลคลาวด์ Supabase ของแอปผักค้าบ const supabase = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_ANON_KEY) // 1. API ดึงข้อมูลผักสด 20 ชนิดไปเรนเดอร์ภาพสีสันหน้าแอปลูกค้า app.get('/api/products', async (req, res) => { try { const { data, error } = await supabase.from('products').select('*') if (error) throw error res.json(data || []) } catch (err) { res.status(500).json({ error: err.message, products: [] }) } }) // 2. API รับบันทึกออเดอร์พรีออเดอร์จากหน้าร้านค้า app.post('/api/orders', async (req, res) => { try { const { customer_name, phone, delivery_address, sub_district, items, total_price, user_tier, discount_amount, earned_points } = req.body const { data, error } = await supabase.from('orders').insert([{ customer_name, phone, delivery_address, sub_district, items, total_price, user_tier, discount_amount, earned_points }]) if (error) throw error res.json({ success: true }) } catch (err) { res.status(500).json({ error: err.message }) } }) // 3. API สำหรับแดชบอร์ดคนขับรถตรวจใบงานออเดอร์ app.get('/api/admin/orders', async (req, res) => { try { const { data, error } = await supabase.from('orders').select('*').order('created_at', { ascending: false }) if (error) throw error res.json(data || []) } catch (err) { res.status(500).json({ error: err.message }) } }) // 4. API อัปเดตสถานะจัดส่งพัสดุหน้าบ้านลูกค้า app.put('/api/admin/orders/:id', async (req, res) => { try { const { id } = req.params const { status } = req.body const { error } = await supabase.from('orders').update({ status }).eq('id', id) if (error) throw error res.json({ success: true }) } catch (err) { res.status(500).json({ error: err.message }) } }) // 5. API ล็อกอินดึงคะแนนสะสมของสมาชิกผ่านเบอร์โทรศัพท์ app.get('/api/member/:phone', async (req, res) => { try { const { phone } = req.params const { data, error } = await supabase.from('orders').select('earned_points').eq('phone', phone).eq('user_tier', 'vip') if (error) throw error const totalPoints = data ? data.reduce((sum, item) => sum + (item.earned_points || 0), 0) : 0 res.json({ points: totalPoints }) } catch (err) { res.json({ points: 0 }) } }) // 6. API ทะเบียนรายชื่อฐานข้อมูลสมาชิกส่วนแอดมิน app.get('/api/admin/members', async (req, res) => { try { const { data, error } = await supabase.from('orders').select('phone, customer_name, user_tier, earned_points').eq('user_tier', 'vip') if (error) throw error const memberMap = {} data.forEach(o => { if (!memberMap[o.phone]) { memberMap[o.phone] = { phone: o.phone, name: o.customer_name, tier: o.user_tier, points: 0 } } memberMap[o.phone].points += (o.earned_points || 0) }) res.json(Object.values(memberMap)) } catch (err) { res.json([]) } }) // 7. API ประมวลผลสถิติสแกนยอดขายธุรกิจรายวัน/เดือน/ปี app.get('/api/admin/analytics', async (req, res) => { try { const { data: orders, error } = await supabase.from('orders').select('*') if (error) throw error const now = new Date() let salesDay = 0, salesMonth = 0, salesYear = 0 const veggieCount = { week: {}, month: {}, year: {} } orders.forEach(order => { const oDate = new Date(order.created_at) const price = Number(order.total_price || 0) const diffDays = Math.ceil(Math.abs(now.getTime() - oDate.getTime()) / (1000 * 60 * 60 * 24)) if (oDate.toDateString() === now.toDateString()) salesDay += price if (oDate.getMonth() === now.getMonth() && oDate.getFullYear() === now.getFullYear()) salesMonth += price if (oDate.getFullYear() === now.getFullYear()) salesYear += price const items = order.items || {} Object.keys(items).forEach(pId => { const qty = Number(items[pId] || 0) if (diffDays <= 7) veggieCount.week[pId] = (veggieCount.week[pId] || 0) + qty if (diffDays <= 30) veggieCount.month[pId] = (veggieCount.month[pId] || 0) + qty if (diffDays <= 365) veggieCount.year[pId] = (veggieCount.year[pId] || 0) + qty }) }) res.json({ salesDay, salesMonth, salesYear, veggieCount }) } catch (err) { res.json({ salesDay: 0, salesMonth: 0, salesYear: 0, veggieCount: { week: {}, month: {}, year: {} } }) } }) // สั่งเปิดพอร์ตระบบ const PORT = process.env.PORT || 3000 app.listen(PORT, () => console.log(`🚀 แอปผักค้าบเซิร์ฟเวอร์วิเคราะห์พร้อมรันแล้วที่พอร์ต ${PORT}`))
